import torch
import matplotlib.pyplot as plt

T = 300  # number of noise steps — enough for good quality on tiny 28x28 images without being slow on CPU

# beta = how much noise gets added at each individual step (small at the start, larger later)
betas = torch.linspace(1e-4, 0.02, T)
alphas = 1.0 - betas
alpha_bars = torch.cumprod(alphas, dim=0)  # cumulative product — lets us jump to any timestep directly, no need to loop

def add_noise(x0, t, noise=None):
    """Take a real image x0 and return it noised to timestep t, in ONE step, using
    the closed-form shortcut instead of adding a little noise 300 times in a loop."""
    if noise is None:
        noise = torch.randn_like(x0)
    sqrt_alpha_bar = alpha_bars[t].sqrt().view(-1, 1, 1, 1)
    sqrt_one_minus_alpha_bar = (1 - alpha_bars[t]).sqrt().view(-1, 1, 1, 1)
    return sqrt_alpha_bar * x0 + sqrt_one_minus_alpha_bar * noise, noise

# Grab one real digit and watch it degrade at increasing timesteps
x0, label = train_data[0]
x0 = x0.unsqueeze(0)  # add batch dimension: [1, 1, 28, 28]

timesteps_to_show = [0, 50, 100, 150, 200, 299]
fig, axes = plt.subplots(1, len(timesteps_to_show), figsize=(15, 3))
for ax, t in zip(axes, timesteps_to_show):
    t_tensor = torch.tensor([t])
    noisy_x, _ = add_noise(x0, t_tensor)
    ax.imshow(noisy_x[0, 0], cmap="gray")
    ax.set_title(f"t={t}")
    ax.axis("off")
plt.show()

import torch.nn as nn
import math

class TimeEmbedding(nn.Module):
    """Turns a single integer timestep (like 137) into a vector the network
    can use — same sinusoidal trick used for positional embeddings in
    Transformers, so the network can tell 'barely noisy' apart from
    'almost pure static' at every layer."""
    def __init__(self, dim=32):
        super().__init__()
        self.dim = dim
        self.mlp = nn.Sequential(nn.Linear(dim, 128), nn.SiLU(), nn.Linear(128, 128))

    def forward(self, t):
        half = self.dim // 2
        freqs = torch.exp(-math.log(10000) * torch.arange(half, device=t.device) / half)
        args = t[:, None].float() * freqs[None, :]
        emb = torch.cat([torch.sin(args), torch.cos(args)], dim=-1)
        return self.mlp(emb)

class ConvBlock(nn.Module):
    def __init__(self, in_ch, out_ch, time_dim=128):
        super().__init__()
        self.conv1 = nn.Conv2d(in_ch, out_ch, 3, padding=1)
        self.conv2 = nn.Conv2d(out_ch, out_ch, 3, padding=1)
        self.norm1 = nn.GroupNorm(8, out_ch)
        self.norm2 = nn.GroupNorm(8, out_ch)
        self.time_proj = nn.Linear(time_dim, out_ch)
        self.act = nn.SiLU()
        self.skip = nn.Conv2d(in_ch, out_ch, 1) if in_ch != out_ch else nn.Identity()

    def forward(self, x, t_emb):
        h = self.act(self.norm1(self.conv1(x)))
        h = h + self.time_proj(t_emb)[:, :, None, None]  # inject "how noisy is this" into every pixel
        h = self.act(self.norm2(self.conv2(h)))
        return h + self.skip(x)  # residual connection

class SimpleUNet(nn.Module):
    def __init__(self):
        super().__init__()
        self.time_embed = TimeEmbedding()

        self.down1 = ConvBlock(1, 32)
        self.down2 = ConvBlock(32, 64)
        self.pool = nn.MaxPool2d(2)

        self.bottleneck = ConvBlock(64, 128)

        self.up1 = nn.ConvTranspose2d(128, 64, 2, stride=2)
        self.up_block1 = ConvBlock(128, 64)  # 128 = 64 (upsampled) + 64 (skip connection)
        self.up2 = nn.ConvTranspose2d(64, 32, 2, stride=2)
        self.up_block2 = ConvBlock(64, 32)   # 64 = 32 (upsampled) + 32 (skip connection)

        self.out = nn.Conv2d(32, 1, 1)

    def forward(self, x, t):
        t_emb = self.time_embed(t)

        d1 = self.down1(x, t_emb)          # 28x28, 32 channels
        d2 = self.down2(self.pool(d1), t_emb)  # 14x14, 64 channels
        b = self.bottleneck(self.pool(d2), t_emb)  # 7x7, 128 channels

        u1 = self.up1(b)                                   # 14x14, 64 channels
        u1 = self.up_block1(torch.cat([u1, d2], dim=1), t_emb)  # reuse d2's detail
        u2 = self.up2(u1)                                  # 28x28, 32 channels
        u2 = self.up_block2(torch.cat([u2, d1], dim=1), t_emb)  # reuse d1's detail

        return self.out(u2)  # predicted noise, same shape as input

unet = SimpleUNet()

from torch.utils.data import DataLoader, Subset

subset_size = 5000
train_subset = Subset(train_data, range(subset_size))
val_subset = Subset(test_data, range(1000))  # MNIST's own test set doubles as held-out validation

train_loader = DataLoader(train_subset, batch_size=64, shuffle=True)
val_loader = DataLoader(val_subset, batch_size=64, shuffle=False)

optimizer = torch.optim.Adam(unet.parameters(), lr=1e-3)
loss_fn = torch.nn.MSELoss()

def run_epoch(loader, train=True):
    unet.train() if train else unet.eval()
    total_loss = 0
    for images, _ in loader:
        t = torch.randint(0, T, (images.shape[0],))  # a random noise-amount per image in the batch
        noisy_images, true_noise = add_noise(images, t)

        predicted_noise = unet(noisy_images, t)
        loss = loss_fn(predicted_noise, true_noise)

        if train:
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()

        total_loss += loss.item()
    return total_loss / len(loader)

best_val_loss = float("inf")
epochs = 10

for epoch in range(epochs):
    train_loss = run_epoch(train_loader, train=True)
    with torch.no_grad():
        val_loss = run_epoch(val_loader, train=False)

    print(f"epoch {epoch+1}: train loss {train_loss:.4f}, val loss {val_loss:.4f}")

    if val_loss < best_val_loss:
        best_val_loss = val_loss
        torch.save(unet.state_dict(), "best_unet.pt")
        print("  -> saved new best model")

@torch.no_grad()
def generate_samples(n=8):
    x = torch.randn(n, 1, 28, 28)  # start from pure noise — this is the ENTIRE input, no image involved at all

    for t in reversed(range(T)):
        t_batch = torch.full((n,), t, dtype=torch.long)
        predicted_noise = unet(x, t_batch)

        alpha_t = alphas[t]
        alpha_bar_t = alpha_bars[t]
        beta_t = betas[t]

        # remove the predicted noise, scaled correctly for this step
        x = (1 / alpha_t.sqrt()) * (x - (beta_t / (1 - alpha_bar_t).sqrt()) * predicted_noise)

        if t > 0:
            noise = torch.randn_like(x)
            x = x + beta_t.sqrt() * noise  # re-add a little fresh randomness, except on the very last step

    return x

import torch.nn as nn

class Generator(nn.Module):
    """Takes pure random noise (100 numbers) and grows it up into a 28x28
    image in one shot — no repeated steps like the diffusion model needed."""
    def __init__(self, noise_dim=100):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(noise_dim, 128 * 7 * 7),
            nn.BatchNorm1d(128 * 7 * 7),
            nn.ReLU(),
            nn.Unflatten(1, (128, 7, 7)),
            nn.ConvTranspose2d(128, 64, 4, stride=2, padding=1),  # 7x7 -> 14x14
            nn.BatchNorm2d(64),
            nn.ReLU(),
            nn.ConvTranspose2d(64, 1, 4, stride=2, padding=1),    # 14x14 -> 28x28
            nn.Tanh(),  # matches the -1..1 range real images will be scaled to
        )

    def forward(self, z):
        return self.net(z)


class Discriminator(nn.Module):
    """Looks at an image (real or fake) and outputs one number: how confident
    it is that the image is real."""
    def __init__(self):
        super().__init__()
        self.net = nn.Sequential(
            nn.Conv2d(1, 64, 4, stride=2, padding=1),   # 28x28 -> 14x14
            nn.LeakyReLU(0.2),
            nn.Conv2d(64, 128, 4, stride=2, padding=1),  # 14x14 -> 7x7
            nn.BatchNorm2d(128),
            nn.LeakyReLU(0.2),
            nn.Flatten(),
            nn.Linear(128 * 7 * 7, 1),
            nn.Sigmoid(),  # outputs a probability: 1 = "I think this is real"
        )

    def forward(self, x):
        return self.net(x)

generator = Generator()
discriminator = Discriminator()

gen_params = sum(p.numel() for p in generator.parameters())
disc_params = sum(p.numel() for p in discriminator.parameters())
print(f"Generator parameters: {gen_params:,}")
print(f"Discriminator parameters: {disc_params:,}")

disc_output = discriminator(fake_images)

from torchvision import transforms
from torch.utils.data import DataLoader

# reload with -1..1 scaling to match the Generator's Tanh output
gan_transform = transforms.Compose([transforms.ToTensor(), transforms.Normalize((0.5,), (0.5,))])
gan_train_data = datasets.MNIST(root="./data", train=True, transform=gan_transform)
gan_loader = DataLoader(gan_train_data, batch_size=64, shuffle=True)

generator = Generator()
discriminator = Discriminator()

loss_fn = nn.BCELoss()
opt_g = torch.optim.Adam(generator.parameters(), lr=2e-4, betas=(0.5, 0.999))
opt_d = torch.optim.Adam(discriminator.parameters(), lr=2e-4, betas=(0.5, 0.999))

epochs = 10

for epoch in range(epochs):
    g_losses, d_losses = [], []

    for real_images, _ in gan_loader:
        batch_size = real_images.shape[0]
        real_labels = torch.ones(batch_size, 1)
        fake_labels = torch.zeros(batch_size, 1)

        # --- train the Discriminator: tell real from fake ---
        noise = torch.randn(batch_size, 100)
        fake_images = generator(noise).detach()  # detach — don't update Generator during this step

        d_loss_real = loss_fn(discriminator(real_images), real_labels)
        d_loss_fake = loss_fn(discriminator(fake_images), fake_labels)
        d_loss = d_loss_real + d_loss_fake

        opt_d.zero_grad()
        d_loss.backward()
        opt_d.step()

        # --- train the Generator: try to fool the Discriminator ---
        noise = torch.randn(batch_size, 100)
        fake_images = generator(noise)
        g_loss = loss_fn(discriminator(fake_images), real_labels)  # wants the Discriminator to say "real"

        opt_g.zero_grad()
        g_loss.backward()
        opt_g.step()

        g_losses.append(g_loss.item())
        d_losses.append(d_loss.item())

    print(f"epoch {epoch+1}: generator loss {sum(g_losses)/len(g_losses):.3f}, discriminator loss {sum(d_losses)/len(d_losses):.3f}")
